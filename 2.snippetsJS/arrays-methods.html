The Array Show

[dotNotation], push, unshift, splice

Access (dotNotation)
Ex: var fruits = ['apples', 'carrots', 'potatoes', 'pears'];
> fruits[0]
> 'apples'     //enough said
Add (push, unshift)
> fruits.push('banana');
> 5     //adds banana to the end of array
> fruits.unshift('oranges');     //add to the beginning of array
> 4     //returns array length
Splice
> fruits = ['apples', 'oranges', 'pears'];
  //Insert at index 1 a new element
> fruits.splice('1','0','blueberries');   //1 position,0 nothing to removed > []
> fruits;
> ['apples', 'blueberries', 'oranges', 'pears']

// remove pears with slice method
> fruits.splice(3,1);
> ["pears"];
> fruits;
> ['apples', 'carrots', 'potatoes']

// remove an element and replace it with another one
// remove potatoes and replace it with avocato
> fruits.splice(2,1,'avocato'); > 'potatoes'     //returns removed elem
> fruits
> ['apples', 'carrots', 'avocato']
indexOf, pop(), shift()
> fruits = ["bananas", "apples", "oranges" "lemons"];
> fruits.indexOf("bananas");     //returns index position
> 0
// remove the last element from an array > fruits.pop();
> "lemons"
> fruits;
> ["bananas", "apples", "oranges"]

// remove first element from array
> fruits.shift();
> "bananas"
> fruits;
> ["apples, "oranges"]
